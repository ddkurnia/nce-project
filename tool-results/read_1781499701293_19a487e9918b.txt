  1440→    function drawMiniChart(canvasId, up) {
  1441→      const canvas = document.getElementById(canvasId);
  1442→      if (!canvas) return;
  1443→      const ctx = canvas.getContext('2d');
  1444→      const w = 48, h = 24;
  1445→      canvas.width = w * 2; canvas.height = h * 2;
  1446→      ctx.scale(2, 2);
  1447→      const points = [];
  1448→      let y = h / 2;
  1449→      for (let i = 0; i < 8; i++) {
  1450→        y += (Math.random() - (up ? 0.35 : 0.65)) * 5;
  1451→        y = Math.max(3, Math.min(h - 3, y));
  1452→        points.push({ x: (i / 7) * w, y });
  1453→      }
  1454→      ctx.beginPath();
  1455→      ctx.moveTo(points[0].x, points[0].y);
  1456→      for (let i = 1; i < points.length; i++) {
  1457→        const cp1x = (points[i - 1].x + points[i].x) / 2;
  1458→        ctx.bezierCurveTo(cp1x, points[i - 1].y, cp1x, points[i].y, points[i].x, points[i].y);
  1459→      }
  1460→      ctx.strokeStyle = up ? '#10b981' : '#ef4444';
  1461→      ctx.lineWidth = 1.5;
  1462→      ctx.stroke();
  1463→    }

  1490→        points.push({ x: (i / 29) * w, y });
  1491→      }
  1492→      ctx.beginPath();
  1493→      ctx.moveTo(points[0].x, points[0].y);
  1494→      for (let i = 1; i < points.length; i++) {
  1495→        const cp1x = (points[i - 1].x + points[i].x) / 2;
